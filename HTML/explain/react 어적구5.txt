import { useReducer } from 'react';
useReducer(함수, 초깃값)
const [state, dispatch] = useReducer(함수, 초깃값) //dispatch가 useReucer안에 있는 함수를 실행시킨다.

action {
	type: "",
}
dispatch({type: "INCREASE"}) //action을 발생시키는 함수

function reducer(state, action){
	switch (action.type){
	case 'INCREASE':
	return state+1;
	case 'DECREASE':
	return state-1;
	}
}
