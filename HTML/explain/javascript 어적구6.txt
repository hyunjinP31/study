1. while
let j = 0;
while(조건){
	조건이 true면 실행
}
do {
	조건이 true면 실행
}while(조건:표현식);
조건이 맞든 아니든 일단 한 번은 실행. 그 뒤 조건이 맞지 않으면 중단. 맞으면 계속함.

반복문에 증감을 안 주면 계속 반복됨(무한루프) 그럼 다운될 수도 있으니까 주의

2. for
for(초기값; 조건; 증감){
	실행할 구문
}
ex> for(let i = 0; i<10; i++){
	console.log(i)
}


반복문 빠져나오기
1. break : 완전히 빠져나오기
for(let i=0; i<10; i++){
	console.log(i);
	if(i===6){
		break(이 반복을 완전히 멈춰버림) continue(6을 건너뛰고 다음 반복을 실행);
	}
}
2. continue : 다음 반복으로 넘어가기

삼항조건식에 continue 혹은 break 사용 불가

let arr = ['사과','딸기','바나나','오렌지'];
(let arr = ['사과',10,'바나나',true]; - 다른 타입의 데이터를 넣을 수도 있다. 하지만 잘 쓰지 않음)
arr[0] 사과가 나옴
arr[1] 딸기가 나옴
arr[2] 바나나 나옴
배열 값마다 인덱스가 매겨져 있고 첫 번째 값은 0. 그 이후 순차적으로 올라간다.

값을 비워놓고 빈 배열을 만들 수도 있으며, 이런 빈 배열의 Boolean값은 true이다.

length 길이
arr.length : arr라는 배열에 담긴 '요소' 개수 그냥 string에 쓸땐 음절 개수
length-1 = 배열의 마지막 요소. << 무족권

arr[arr.length]='키위'
arr = ['귤','사과','오렌지','키위']마지막 요소를 추가할 수 있음
